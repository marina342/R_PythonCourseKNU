> my_ifelse <- function (x, exp, n) {
     if(exp == "<") {
         res <- which(x < n)
     }
     else if(exp == ">") {
         res <- which(x > n)
     }
     else if(exp == "<="){
         res <- which(x <= n)
     }
     else if(exp == ">="){
         res <- which(x >= n)
     }

     }
